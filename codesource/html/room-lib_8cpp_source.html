98 // 🔒 Enforce key size here
99 if (keyLen != 16 && keyLen != 32) {
100 jclass ex = env->FindClass("java/lang/IllegalArgumentException");
101 env->ThrowNew(ex, "Invalid key length. Only 128-bit or 256-bit keys are supported.");
102 return nullptr;
103 }
104
105 const char *chars = env->GetStringUTFChars(passphrase, nullptr);
106 std::string password(chars);
107 env->ReleaseStringUTFChars(passphrase, chars);
108
109 jsize saltLen = env->GetArrayLength(salt);
110 std::vector<uint8_t> saltVec(saltLen);
111 env->GetByteArrayRegion(salt, 0, saltLen, reinterpret_cast<jbyte *>(saltVec.data()));
112
113 std::vector<uint8_t> key = room::pbkdf2_sha256(password, saltVec, iterations, keyLen);
114
115 jbyteArray result = env->NewByteArray(keyLen);
116 env->SetByteArrayRegion(result, 0, keyLen, reinterpret_cast<jbyte *>(key.data()));
117 return result;
