26 // ---------- Key validation ----------
27
28 bool is_valid_aes_key_size(size_t keySize) {
29 return keySize == 16 || keySize == 32; // 128-bit or 256-bit
30 }
31
32 const EVP_CIPHER* select_cipher(size_t keySize) {
33 if (keySize == 16) return EVP_aes_128_gcm();
34 if (keySize == 32) return EVP_aes_256_gcm();
35 return nullptr;
36 }

40 std::string encrypt_string(const std::string& plaintext, const std::string& keyHex) {
41
42 std::vector<uint8_t> key = shared::from_hex(keyHex);
43
44 if (!is_valid_aes_key_size(key.size())) {
45 throw std::invalid_argument(
46 "Invalid AES key size. Only 128-bit or 256-bit keys are supported."
47 );
48 }
49
50 const EVP_CIPHER* cipher = select_cipher(key.size());
51 if (!cipher) {
52 throw std::runtime_error("Failed to select AES cipher");
53 }
54
55 std::vector<uint8_t> iv(IV_SIZE);
56 RAND_bytes(iv.data(), safe_size_cast(iv));
57
58 std::vector<uint8_t> ciphertext(plaintext.size());
59 std::vector<uint8_t> tag(TAG_SIZE);
60
61 EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
62 EVP_EncryptInit_ex(ctx, cipher, nullptr, nullptr, nullptr);
63 EVP_EncryptInit_ex(ctx, nullptr, nullptr, key.data(), iv.data());
64
65 int len;
66 EVP_EncryptUpdate(
67 ctx,
68 ciphertext.data(),
69 &len,
70 reinterpret_cast<const uint8_t*>(plaintext.data()),
71 safe_size_cast(plaintext.size())
72 );
73
74 int ciphertext_len = len;
75 EVP_EncryptFinal_ex(ctx, ciphertext.data() + len, &len);
76 ciphertext_len += len;
77 ciphertext.resize(ciphertext_len);
78
79 EVP_CIPHER_CTX_ctrl(ctx, EVP_CTRL_GCM_GET_TAG, TAG_SIZE, tag.data());
80 EVP_CIPHER_CTX_free(ctx);
81
82 std::ostringstream oss;
83 oss << shared::to_hex(iv)
84 << ":" << shared::to_hex(ciphertext)
85 << ":" << shared::to_hex(tag);
86
87 return oss.str();
88 }

92 std::string decrypt_string(const std::string& input, const std::string& keyHex) {
93
94 std::vector<uint8_t> key = shared::from_hex(keyHex);
95
96 if (!is_valid_aes_key_size(key.size())) {
97 throw std::invalid_argument(
98 "Invalid AES key size. Only 128-bit or 256-bit keys are supported."
99 );
100 }
101
102 const EVP_CIPHER* cipher = select_cipher(key.size());
103 if (!cipher) {
104 throw std::runtime_error("Failed to select AES cipher");
105 }
106
107 size_t p1 = input.find(':');
108 size_t p2 = input.find(':', p1 + 1);
109
110 auto iv = shared::from_hex(input.substr(0, p1));
111 auto ciphertext = shared::from_hex(input.substr(p1 + 1, p2 - p1 - 1));
112 auto tag = shared::from_hex(input.substr(p2 + 1));
113
114 EVP_CIPHER_CTX* ctx = EVP_CIPHER_CTX_new();
115 EVP_DecryptInit_ex(ctx, cipher, nullptr, nullptr, nullptr);
116 EVP_DecryptInit_ex(ctx, nullptr, nullptr, key.data(), iv.data());
117
118 std::vector<uint8_t> plaintext(ciphertext.size());
119
120 int len = 0;
121 int plaintext_len = 0;
122
123 EVP_DecryptUpdate(
124 ctx,
125 plaintext.data(),
126 &len,
127 ciphertext.data(),
128 safe_size_cast(ciphertext.size())
129 );
130
131 plaintext_len = len;
132
133 __android_log_print(ANDROID_LOG_DEBUG, "RoomNative", "🔐 GCM tag verification");
134
135 EVP_CIPHER_CTX_ctrl(
136 ctx,
137 EVP_CTRL_GCM_SET_TAG,
138 TAG_SIZE,
139 const_cast<uint8_t*>(tag.data())
140 );
141
142 // 🔑 Step 3: final verification
143 int ret = EVP_DecryptFinal_ex(
144 ctx,
145 plaintext.data() + plaintext_len,
146 &len
147 );
148
149 __android_log_print(ANDROID_LOG_DEBUG, "RoomNative", "🔐 GCM tag verification: %d", ret);
150
151 EVP_CIPHER_CTX_free(ctx);
152
153 if (ret <= 0) {
154 __android_log_print(ANDROID_LOG_ERROR, "RoomNative", "🔐 GCM tag verification failed");
155 return "";
156 }
157
158 plaintext_len += len;
159
160 return {
161 reinterpret_cast<char*>(plaintext.data()),
162 static_cast<std::string::size_type>(plaintext_len)
163 };
164 }
